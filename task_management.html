        `;
        document.body.appendChild(taskDetailsModal);

        const closeTaskDetails = document.getElementById('closeTaskDetails');
        const closeTaskDetailsBtn = document.getElementById('closeTaskDetailsBtn');

        closeTaskDetails.addEventListener('click', () => {
            taskDetailsModal.style.display = 'none';
        });

        closeTaskDetailsBtn.addEventListener('click', () => {
            taskDetailsModal.style.display = 'none';
        });

        taskDetailsModal.addEventListener('click', (e) => {
            if (e.target === taskDetailsModal) {
                taskDetailsModal.style.display = 'none';
            }
        });

        // 为表格中的"查看"链接添加事件监听
        taskTable.addEventListener('click', (e) => {
            if (e.target.textContent.includes('查看')) {
                const row = e.target.closest('tr');
                const taskId = row.cells[1].textContent;
                const taskName = row.cells[2].textContent;
                const taskType = row.cells[3].textContent;
                const dataSource = row.cells[4].textContent;
                const createTime = row.cells[5].textContent;
                const endTime = row.cells[6].textContent;
                const status = row.cells[7].textContent;
                const progressBar = row.cells[8].querySelector('.progress-bar-fill');
                const progressWidth = progressBar ? progressBar.style.width : '0%';
                const priority = row.cells[9].textContent;

                // 填充详情模态框
                document.getElementById('detailTaskId').textContent = taskId;
                document.getElementById('detailTaskName').textContent = taskName;
                document.getElementById('detailTaskType').textContent = taskType;
                document.getElementById('detailTaskPriority').textContent = priority;
                document.getElementById('detailCreateTime').textContent = createTime;
                document.getElementById('detailEndTime').textContent = endTime;
                document.getElementById('detailDataSource').textContent = dataSource;
                document.getElementById('detailStatus').textContent = status;
                document.getElementById('detailProgressBar').style.width = progressWidth;
                document.getElementById('detailTaskDescription').textContent = '这是任务的详细描述内容...';
                document.getElementById('detailTaskLog').textContent = '[' + createTime + '] 任务开始\n[' + new Date().toLocaleString() + '] 处理中...\n';

                taskDetailsModal.style.display = 'flex';
            }
        });
